#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(int argc, char *argv[])
{
    int rank, size;
    int n;                  // Total elements (user input)
    int key;                // Key to search
    int *array = NULL;      // Full array (master only)
    int *sub_array = NULL;  // Sub-array for each process
    int sub_n;              // Elements per process
    int local_count = 0;
    int global_count = 0;

    double start_time, end_time, local_time;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    /* -------- MASTER TAKES INPUT -------- */
    if (rank == 0)
    {
        printf("Enter number of elements: ");
        fflush(stdout);
        scanf("%d", &n);

        if (n % size != 0)
        {
            printf("Number of elements must be divisible by number of processes!\n");
            MPI_Abort(MPI_COMM_WORLD, 1);
        }

        array = (int *)malloc(n * sizeof(int));

        srand(time(NULL));

        printf("\nGenerated Array:\n");
        for (int i = 0; i < n; i++)
        {
            array[i] = rand() % 10;
            printf("%d ", array[i]);
        }
        printf("\n");

        printf("\nEnter the key to find frequency: ");
        fflush(stdout);
        scanf("%d", &key);
    }

    /* Broadcast n and key to all processes */
    MPI_Bcast(&n, 1, MPI_INT, 0, MPI_COMM_WORLD);
    MPI_Bcast(&key, 1, MPI_INT, 0, MPI_COMM_WORLD);

    sub_n = n / size;
    sub_array = (int *)malloc(sub_n * sizeof(int));

    /* Scatter array */
    MPI_Scatter(array, sub_n, MPI_INT,
                sub_array, sub_n, MPI_INT,
                0, MPI_COMM_WORLD);

    /* -------- EACH PROCESS WORK -------- */
    MPI_Barrier(MPI_COMM_WORLD);  // Synchronize before timing

    start_time = MPI_Wtime();

    for (int i = 0; i < sub_n; i++)
    {
        if (sub_array[i] == key)
            local_count++;
    }

    end_time = MPI_Wtime();
    local_time = end_time - start_time;

    /* Print each process details in order */
    for (int i = 0; i < size; i++)
    {
        if (rank == i)
        {
            printf("\nProcess %d:\n", rank);
            printf("Sub-array: ");
            for (int j = 0; j < sub_n; j++)
                printf("%d ", sub_array[j]);

            printf("\nLocal Frequency of %d = %d", key, local_count);
            printf("\nTime taken = %f seconds\n", local_time);
        }
        MPI_Barrier(MPI_COMM_WORLD);
    }

    /* Reduce to get global count */
    MPI_Reduce(&local_count, &global_count, 1, MPI_INT,
               MPI_SUM, 0, MPI_COMM_WORLD);

    if (rank == 0)
    {
        printf("\n--------------------------------");
        printf("\nTotal Frequency of %d = %d\n", key, global_count);
        printf("--------------------------------\n");
    }

    free(sub_array);
    if (rank == 0)
        free(array);

    MPI_Finalize();
    return 0;
}
