#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <omp.h>

#define MAX 100
#define MAX_LEN 256

char queue[MAX][MAX_LEN];
int count = 0;

omp_lock_t lock;

int main() {

    omp_init_lock(&lock);

    double global_start = omp_get_wtime();

    #pragma omp parallel num_threads(4)
    {
        int tid = omp_get_thread_num();
        double t_start = omp_get_wtime();

        // PRODUCERS
        if (tid < 2) {
            FILE *fp;
            char filename[20];
            char line[MAX_LEN];

            sprintf(filename, "file%d.txt", tid + 1);
            fp = fopen(filename, "r");

            if (fp != NULL) {
                while (fgets(line, MAX_LEN, fp)) {

                    omp_set_lock(&lock);

                    if (count < MAX) {
                        strcpy(queue[count], line);

                        #pragma omp atomic
                        count++;

                        printf("Producer Thread %d inserted: %s", tid, line);
                    }

                    omp_unset_lock(&lock);
                }
                fclose(fp);
            } else {
                printf("Thread %d: File not found\n", tid);
            }
        }

        // CONSUMERS
        else {
            while (1) {
                char line[MAX_LEN];

                omp_set_lock(&lock);

                if (count > 0) {
                    strcpy(line, queue[count - 1]);

                    #pragma omp atomic
                    count--;

                    omp_unset_lock(&lock);

                    printf("Consumer Thread %d processing line: %s", tid, line);

                    // Tokenization
                    char *token = strtok(line, " \t\n");
                    while (token != NULL) {

                        #pragma omp critical
                        {
                            printf("Thread %d token: %s\n", tid, token);
                        }

                        token = strtok(NULL, " \t\n");
                    }
                } else {
                    omp_unset_lock(&lock);
                    break;
                }
            }
        }

        double t_end = omp_get_wtime();

        #pragma omp critical
        {
            printf("Thread %d finished. Time taken = %f seconds\n",
                   tid, t_end - t_start);
        }
    }

    double global_end = omp_get_wtime();

    printf("Total Execution Time = %f seconds\n", global_end - global_start);

    omp_destroy_lock(&lock);

    return 0;
}
