#include <stdio.h>
#include <omp.h>

double f(double x) {
    return x * x;
}

int main() {
    int n = 1000000;
    double a = 0.0, b = 1.0;
    double h = (b - a) / n;
    double sum = 0.0;

    double start_time = omp_get_wtime();

    #pragma omp parallel
    {
        int tid = omp_get_thread_num();
        double local_sum = 0.0;
        double t_start = omp_get_wtime();

        #pragma omp for
        for (int i = 1; i < n; i++) {
            double x = a + i * h;
            local_sum += f(x);
        }

        #pragma omp atomic
        sum += local_sum;

        double t_end = omp_get_wtime();

        #pragma omp critical
        {
            printf("Thread %d computed partial sum = %f, Time = %f seconds\n",
                   tid, local_sum, t_end - t_start);
        }
    }

    double result;

    #pragma omp critical
    {
        result = h * ((f(a) + f(b)) / 2.0 + sum);
    }

    double end_time = omp_get_wtime();

    printf("Final Integral = %f\n", result);
    printf("Total Execution Time = %f seconds\n", end_time - start_time);

    return 0;
}
